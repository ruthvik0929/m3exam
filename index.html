#include <Servo.h> 
Servo doorServo; 
const int servoPin = 9; 
const int LED_PIN = 13; 
void setup() { 
doorServo.attach(servoPin); 
doorServo.write(0); 
pinMode(LED_PIN, OUTPUT); 
Serial.begin(9600); 
} 
void loop() { 
if (Serial.available() > 0) { 
char command = Serial.read(); 
if (command == '1') { 
doorServo.write(90); 
digitalWrite(LED_PIN, HIGH); 
}
else if(command == '0') {
  doorServo.write(0);
  digitalWrite(LED_PIN,LOW);
}
}
}
