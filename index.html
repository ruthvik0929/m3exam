Rows:
n=int(input("enter no. of rows"))
for a in range(n):
    for b in range(n-a-1):
        print(" ",end="")
    for b in range(a+1):
        print(a+1,end=" ")
    print()

Bar Graph:
%matplotlib inline
import matplotlib.pyplot as plt
categories = ['Math', 'Science', 'History', 'Art']
scores = [85, 90, 78, 88]
plt.bar(categories, scores, color=['blue', 'green', 'red', 'purple'])
plt.xlabel('Subjects')
plt.ylabel('Scores')
plt.title('Student Scores in Different Subjects')
plt.show()
